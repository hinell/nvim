Hinell Neovim config

This folder features my config for great Neovim editor.

Usage

lua require("init")
lua <<EOF
	-- local cwd = vim.fn.getcwd()
	--Set up custom module searching path for user (hinell)
	local	userPath = vim.fn.stdpath("config")
  			package.path = ";;" .. userPath .. "/?/init.lua;" .. userPath .. "/hinell/?.lua;;"
            require("hinell")
EOF
